  flags?: Set<string>;

}

  <files>src/services/message-service.ts</files>
  <behavior>
    - listMessages(client, folder, opts) where opts: { limit?: number; offset?: number; sort?: 'newest' | 'oldest' }
    - Acquires getMailboxLock(folder, { readOnly: true }) before any fetch
    - Calls search({ all: true }, { uid: true }) — normalizes false to [] immediately
    - Sorts descending for 'newest' (default), ascending for 'oldest'
    - Slices: uids.slice(offset, offset + limit) where limit defaults to 50, offset defaults to 0
    - If pageUids is empty, releases lock and returns []
    - Calls fetchAll(pageUids, { uid: true, envelope: true, flags: true, internalDate: true }, { uid: true })
    - Maps to MessageHeader: uid=msg.uid, from=first from address or '', subject or '', date as ISO string, unread=!flags.has('\\Seen')
    - Lock ALWAYS released in finally block
  </behavior>
  <action>
Create `src/services/message-service.ts`. Implement `listMessages` per behavior block.

Key implementation details:
- `const allUids = (await client.search({ all: true }, { uid: true })) || [];` — note the `|| []` to handle false
- For sort: `sort === 'oldest' ? allUids : [...allUids].sort((a, b) => b - a)` — spread to avoid mutating the original array
- `msg.internalDate instanceof Date ? msg.internalDate.toISOString() : String(msg.internalDate ?? '')` for safe date conversion
- `msg.envelope?.from?.[0]?.address ?? ''` for safe from extraction
- `!msg.flags?.has('\\Seen')` for unread flag

Do NOT catch errors inside the lock block — let them propagate after the finally releases the lock.
